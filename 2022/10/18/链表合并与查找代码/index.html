// ConsoleApplication11.cpp : 此文件包含 "main" 函数。程序执行将在此处开始并结束。
//

#include <iostream>
using namespace std;
struct Linklist {
	long id;
	int data;
	struct Linklist* Next;
};
struct Linklist* CreateList(int dat,long id) {
	struct Linklist* head = (struct Linklist*)malloc(sizeof(struct Linklist));
	head->data = dat;
	head->id = id;
	head->Next = NULL;
	return head;
}  //Create HeadNode
void InsertData_H(struct Linklist* head, int dat, long id) {
	struct Linklist* p = (struct Linklist*)malloc(sizeof(struct Linklist));
	p->data = dat;
	p->id = id;
	p->Next = head->Next;
	head->Next = p;
} //HeadInsert 
struct Linklist* InsertData_T(struct Linklist* tail, int dat, long id) {
	struct Linklist* p = (struct Linklist*)malloc(sizeof(struct Linklist));
	tail->Next = p;
	p->data = dat;
	p->id = id;
	p->Next = NULL;
	tail = p;
	return tail;
}
void SearchLink_Num(struct Linklist* head, int num) {
	struct Linklist* p = head;
	while(p->data != num &&p!=NULL)
	{
		p = p->Next;
	}
	if (p) {
		cout << "The student ID of" << p->data << " is " << p->id << endl;
	}
	else {
		cout << "Can not find the student ID of" << num << endl;
		cout << "Please check your input" << endl;
	}
}
void SearchLink_ID(struct Linklist* head, int id) {
	struct Linklist* p = head;
	while(p->id != id && p != NULL)
	{
		p = p->Next;
	}
	if (p) {
		cout << "The data of ID:" << p->id << " is " << p->data << endl;
	}
	else {
		cout << "Can not find the data of ID:" << id << endl;
		cout << "Please check your input" << endl;
	}
}
struct Linklist* Linkcombine(struct Linklist* head1, struct Linklist* head2) {
	struct Linklist* p1 = head1->Next;
	struct Linklist* q1 = head2->Next;
	struct Linklist* H = NULL;
	head1->id > head2->id ? H = head2 : H = head1;
	while (p1 != NULL && q1 != NULL) {
		if (head1->id < head2->id) {
			while (p1 != NULL&&p1->id < head2->id) {
				head1 = p1;
				p1 = p1->Next;
			}
			head1->Next = head2;
			if (p1) {
				head1 = p1;
				p1 = p1->Next;
			}
		}
		else if (head2->id <= head1->id) {
			while ( q1 != NULL&&q1->id < head2->id) {
				head2 = q1;
				q1 = q1->Next;
			}
			head2->Next = head1;
			if (q1) {
				head2 = q1;
				q1 = q1->Next;
			}
		}
	}
	if (p1 != NULL) {
		while (p1->Next != NULL && p1->id < head2->id) {  //使 p1>head2 ,否则指向倒数第一个和倒数第二个
			head1 = p1;
			p1 = p1->Next;
		}
		if (head2->id < p1->id&& head2->id > head1->id) {
			head2->Next = p1;
			head1->Next = head2;
		}
		else
		{
			p1->Next = head2;
		}
	}
	else if (q1 != NULL) {
		while (q1->Next!=NULL&&q1->id<head1->id) {
			head2 = q1;
			q1 = q1->Next;
		}
		if (head1->id < q1->id && head1->id > head2->id) {
			head1->Next = q1;
			head2->Next = head1;
		}
		else
		{
			q1->Next = head1;
		}
	}
	return H;
}
