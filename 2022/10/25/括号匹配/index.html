括号匹配

2022-10-25 281 words 
#include <iostream>
struct Node {
    char data;
    struct Node* next;
};
struct Node* CreateStack() {
    struct Node * p = (struct Node*)malloc(sizeof(struct Node));
    p->data = 0;
    p->next = NULL;
    return p;
} //空的头结点
void InsertStack(struct Node* s, char str) {
    struct Node* p = (struct Node*)malloc(sizeof(struct Node));
    p->data = str;
    p->next = s->next;
    s->next = p;
}
struct Node* PopStack(struct Node* s) {
    struct Node* p = s->next;
    s->next = p->next;
    return p;
}
bool IsEmpty(struct Node* s) {
    return (s->next == NULL);
}
int judge(char s) {
    if (s == '(' or s == '{' or s == '[')
        return 1;
    else
        return 0;
}
int main()
{
    using namespace std;
    struct Node* p=NULL;
    struct Node* stack = CreateStack();
    char s;
    int flag = 1;
    cout << "input a string(including '([{' and ')]}'\n";
    while ((s = cin.get()) != '\n'&&flag) {
        if (judge(s))
            InsertStack(stack, s);
        else
        {   
            if (!IsEmpty(stack))
                p = PopStack(stack);
            else
            {
                flag = 0;
                break;
            }
            switch (s)
            {
            case ')':
                p->data == '(' ? flag = 1 : flag = 0;
                break;
            case '}':
                p->data == '{' ? flag = 1 : flag = 0;
                break;
            case ']':
                p->data == '[' ? flag = 1 : flag = 0;
                break;
            default:
                cout << "Input wrong string!\n";
                flag = 0;
                break;
            }
        }
    }
    if (flag)
        cout << "matching Successfully\n";
    else
        cout << "matching failed\n";
    return 0;
}
