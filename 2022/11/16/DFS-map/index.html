DFS_map

练手

#include <iostream>
#include <random>
#include <ctime>
#include <cstdlib>
using namespace std;
int map_[10][10] = {0};
bool flag[10] = { false };
int dfs(int i) {
    if (flag[i] == true)return 0;
    else
    {
        flag[i] = true;
        cout << i+1 << " ";
        for (auto j = 0; j < 10; j++) {
            if (map_[i][j] == 1)
                dfs(j);
        }
    }
    return 0;
}
int main()
{
    int i, j = 0;
    srand(time(0));
    for (i = 0; i < 10; i++) {
        for (j = i+1; j < 10; j++) {
            map_[i][j]=rand() % 2;
            map_[j][i] = map_[i][j];
        }
    }
    cout << "用随机数生成的用邻接矩阵表示的无向图:" << endl;
    for (i = 0; i < 10; i++) {
        for (j = 0; j < 10; j++) {
            cout << map_[i][j] << " ";
        }
        cout << endl;
    }
    // initialize the map
    cout << '\n';
    cout << "遍历顺序（从1开始）" << endl;
    dfs(0);
    return 0;
}
